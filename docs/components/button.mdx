## Types

### Text buttons

Text buttons are buttons that use text paired with or without an icon.

```jsx
// preview

import { Button, Icon, Box } from "@washingtonpost/wpds-ui-kit";
import BookmarkSolid from "@washingtonpost/wpds-assets/asset/bookmark-solid";
export default function Example() {
  return (
    <Box
      css={{
        display: "flex",
        justifyContent: "center",
        width: "100%",
        gap: "$100",
      }}
    >
      <Button variant="primary" size="200" icon="left">
        <Icon size="100">
          <BookmarkSolid />
        </Icon>
        Text button
      </Button>
      <Button variant="primary" size="200" icon="right">
        <Icon size="100">
          <BookmarkSolid />
        </Icon>
        Text button
      </Button>
      <Button variant="primary" size="200">
        Text button
      </Button>
    </Box>
  );
}
```

### Icon buttons

An icon button is a button that uses only an icon.

```jsx
// preview

import { Button, Icon, Box } from "@washingtonpost/wpds-ui-kit";
import Add from "@washingtonpost/wpds-assets/asset/add";
export default function Example() {
  return (
    <Box
      css={{
        display: "flex",
        justifyContent: "center",
        width: "100%",
        gap: "$100",
      }}
    >
      <Button variant="primary" size="200" icon="center">
        <Icon size="100">
          <Add />
        </Icon>
      </Button>
    </Box>
  );
}
```

### isOutline

Adding the property `isOutline` will change the appearance of the button to be an outline button.

```jsx
// preview

import { Button, Icon, Box } from "@washingtonpost/wpds-ui-kit";
import Add from "@washingtonpost/wpds-assets/asset/add";
export default function Example() {
  return (
    <Box
      css={{
        display: "flex",
        justifyContent: "center",
        alignItems: "center",
        width: "100%",
        gap: "$100",
      }}
    >
      <Button variant="primary" isOutline size="200" icon="left">
        <Icon size="100">
          <Add />
        </Icon>
        Text button
      </Button>
      <Box css={{ backgroundColor: "$gray20", padding: "$100" }}>
        <Button variant="secondary" isOutline size="200" icon="left">
          <Icon size="100">
            <Add />
          </Icon>
          Text button
        </Button>
      </Box>
      <Button variant="cta" isOutline size="200" icon="left">
        <Icon size="100">
          <Add />
        </Icon>
        Text button
      </Button>
    </Box>
  );
}
```
